"""
Utility functions for Grease Pencil system.

Provides seed generation, validation, and helper functions
for deterministic GP generation.
"""

from typing import Any, Dict, Optional
import hashlib


def generate_seed_from_params(params: Dict[str, Any]) -> int:
    """
    Generate deterministic seed from parameter dictionary.

    Uses SHA-256 hash of sorted params for reproducibility.
    Same params always produce same seed.

    Args:
        params: Dictionary of generation parameters

    Returns:
        Integer seed in range [0, 2^31)

    Example:
        >>> seed1 = generate_seed_from_params({'stroke_count': 10, 'width': 5.0})
        >>> seed2 = generate_seed_from_params({'stroke_count': 10, 'width': 5.0})
        >>> assert seed1 == seed2
    """
    # If seed already provided, use it
    if 'seed' in params and params['seed'] is not None:
        return int(params['seed']) % (2**31)

    # Hash sorted params for consistency
    param_str = str(sorted(params.items(), key=lambda x: x[0]))
    hash_bytes = hashlib.sha256(param_str.encode()).digest()

    # Convert first 4 bytes to integer
    seed = int.from_bytes(hash_bytes[:4], byteorder='big')
    return seed % (2**31)


def validate_gp_params(params: Dict[str, Any]) -> Dict[str, Any]:
    """
    Validate and normalize GP parameters.

    Ensures required fields exist with sensible defaults.

    Args:
        params: Raw parameter dictionary

    Returns:
        Validated parameter dictionary
    """
    validated = dict(params)

    # Set defaults
    validated.setdefault('stroke_count', 10)
    validated.setdefault('layer_count', 3)
    validated.setdefault('stroke_width', 5.0)
    validated.setdefault('material_config', {})

    # Validate ranges
    if validated['stroke_count'] < 1:
        raise ValueError(f"stroke_count must be >= 1, got {validated['stroke_count']}")
    if validated['layer_count'] < 1:
        raise ValueError(f"layer_count must be >= 1, got {validated['layer_count']}")
    if validated['stroke_width'] <= 0:
        raise ValueError(f"stroke_width must be > 0, got {validated['stroke_width']}")

    return validated


def create_default_layer_names(count: int, prefix: str = "Layer") -> list:
    """
    Generate default layer names.

    Args:
        count: Number of layer names to generate
        prefix: Name prefix (default "Layer")

    Returns:
        List of layer names like ["Layer_01", "Layer_02", ...]
    """
    return [f"{prefix}_{i:02d}" for i in range(1, count + 1)]

"""
Stage Pipeline for Grease Pencil generation.

Follows GSD Universal Stage Order:
Normalize → Primary → Secondary → Detail → Output

Each stage is a pure function that transforms input to output.
All procedural generation is deterministic via seed parameters.
"""

from typing import Any, Dict, List, Optional, TYPE_CHECKING
import random

if TYPE_CHECKING:
    import bpy
    from bpy.types import Object as BlenderObject

from .types import GPStrokeConfig, GPLayerConfig, GPMaterialConfig, GPMaskConfig
from .utils import generate_seed_from_params, validate_gp_params, create_default_layer_names


def stage_normalize_gp(params: Dict[str, Any]) -> Dict[str, Any]:
    """
    Stage 1: Normalize and validate GP parameters.

    Converts raw input to canonical form with all required fields.
    Derives seed from params if not provided.

    Args:
        params: Raw generation parameters

    Returns:
        Normalized parameter dictionary with:
        - stroke_count: int
        - layer_count: int
        - seed: int (deterministic)
        - stroke_width: float
        - material_config: dict
        - layer_names: List[str]
        - output_path: str

    Example:
        >>> normalized = stage_normalize_gp({'stroke_count': 20})
        >>> assert 'seed' in normalized
        >>> assert normalized['stroke_count'] == 20
    """
    # Validate and get defaults
    validated = validate_gp_params(params)

    # Generate deterministic seed
    seed = generate_seed_from_params(validated)

    # Create layer names
    layer_names = params.get('layer_names') or create_default_layer_names(
        validated['layer_count'],
        params.get('layer_prefix', 'Layer')
    )

    return {
        'stroke_count': validated['stroke_count'],
        'layer_count': validated['layer_count'],
        'seed': seed,
        'stroke_width': validated['stroke_width'],
        'material_config': validated['material_config'],
        'layer_names': layer_names,
        'output_path': params.get('output_path', ''),
        'use_cyclic': params.get('use_cyclic', False),
        'display_mode': params.get('display_mode', '3DSPACE'),
        'original_params': params,  # Keep for reference
    }


def stage_primary_gp(normalized: Dict[str, Any]) -> Dict[str, Any]:
    """
    Stage 2: Create base GP object with layers and strokes.

    Generates core geometry and layer structure.
    This is the main shape generation stage.

    Args:
        normalized: Output from stage_normalize_gp()

    Returns:
        Dictionary with:
        - All normalized fields
        - strokes: List[GPStrokeConfig]
        - layers: List[GPLayerConfig]
        - materials: List[GPMaterialConfig]

    Note:
        Does NOT create Blender objects - returns configs only.
        Blender creation happens in stage_output_gp.
    """
    # Set random state for determinism
    rng = random.Random(normalized['seed'])

    # Generate stroke configurations
    strokes: List[GPStrokeConfig] = []
    for i in range(normalized['stroke_count']):
        stroke = GPStrokeConfig(
            id=f"stroke_{i:04d}",
            points=_generate_random_points(rng, normalized),
            stroke_width=normalized['stroke_width'],
            hardness=1.0,
            use_cyclic=normalized.get('use_cyclic', False),
            material_index=0,
            display_mode=normalized.get('display_mode', '3DSPACE'),
            seed=rng.randint(0, 2**31),
        )
        strokes.append(stroke)

    # Generate layer configurations
    layers: List[GPLayerConfig] = []
    layer_size = normalized['stroke_count'] // normalized['layer_count']

    for i, layer_name in enumerate(normalized['layer_names']):
        start_idx = i * layer_size
        end_idx = start_idx + layer_size if i < normalized['layer_count'] - 1 else normalized['stroke_count']

        layer = GPLayerConfig(
            id=f"layer_{i:02d}",
            name=layer_name,
            opacity=1.0,
            blend_mode='REGULAR',
            use_lights=False,
            use_mask_layer=False,
            stroke_configs=strokes[start_idx:end_idx],
        )
        layers.append(layer)

    # Generate default material config
    materials: List[GPMaterialConfig] = [
        GPMaterialConfig(
            id="default_material",
            show_stroke=True,
            show_fill=True,
        )
    ]

    return {
        **normalized,
        'strokes': strokes,
        'layers': layers,
        'materials': materials,
    }


def stage_secondary_gp(primary_result: Dict[str, Any]) -> Dict[str, Any]:
    """
    Stage 3: Apply modifications to GP data.

    Layer reordering, visibility, blend modes, and rigging setup.
    Does NOT apply GP modifiers (that's stage_detail_gp).

    Args:
        primary_result: Output from stage_primary_gp()

    Returns:
        Modified result with:
        - layers potentially reordered
        - blend modes applied
        - rigging configs added
    """
    result = dict(primary_result)

    # Apply layer modifications from config
    layer_mods = primary_result.get('layer_modifications', {})

    for layer in result['layers']:
        if layer.id in layer_mods:
            mods = layer_mods[layer.id]
            if 'opacity' in mods:
                layer.opacity = mods['opacity']
            if 'blend_mode' in mods:
                layer.blend_mode = mods['blend_mode']
            if 'use_lights' in mods:
                layer.use_lights = mods['use_lights']

    # Add rigging configuration if specified
    if 'rigging' in primary_result.get('original_params', {}):
        result['rigging_config'] = primary_result['original_params']['rigging']

    return result


def stage_detail_gp(secondary_result: Dict[str, Any]) -> Dict[str, Any]:
    """
    Stage 4: Apply GP modifiers with masking.

    Adds GP_NOISE, GP_SMOOTH, GP_BUILD, etc.
    All effects support masking via GPMaskConfig.

    Args:
        secondary_result: Output from stage_secondary_gp()

    Returns:
        Result with:
        - modifier_configs: List of modifier configurations
        - masks applied to each modifier
    """
    result = dict(secondary_result)

    # Get effect configurations
    effects = secondary_result.get('effects', [])
    modifier_configs = []

    for effect in effects:
        mod_config = {
            'type': effect.get('type', 'GP_NOISE'),
            'name': effect.get('name', 'Effect'),
            'params': effect.get('params', {}),
            'mask': effect.get('mask'),
        }
        modifier_configs.append(mod_config)

    result['modifier_configs'] = modifier_configs

    return result


def stage_output_gp(detail_result: Dict[str, Any]) -> Dict[str, Any]:
    """
    Stage 5: Prepare for final output.

    Creates Blender GP object, applies materials, sets render settings.
    This is where actual Blender objects are created.

    Args:
        detail_result: Output from stage_detail_gp()

    Returns:
        Final result with:
        - gp_object: bpy.types.Object (if in Blender)
        - render_settings: Dict
        - export_config: Dict
    """
    result = dict(detail_result)

    # Prepare render settings
    render_settings = {
        'engine': 'BLENDER_EEVEE_NEXT',
        'frame_start': detail_result.get('frame_start', 1),
        'frame_end': detail_result.get('frame_end', 100),
        'resolution_x': detail_result.get('resolution_x', 1920),
        'resolution_y': detail_result.get('resolution_y', 1080),
    }
    result['render_settings'] = render_settings

    # Prepare export config
    export_config = {
        'format': detail_result.get('export_format', 'PNG'),
        'output_path': detail_result.get('output_path', ''),
    }
    result['export_config'] = export_config

    # Note: Actual Blender object creation happens here when run in Blender
    # For testing without Blender, we skip this
    try:
        import bpy
        result['_blender_available'] = True
    except ImportError:
        result['_blender_available'] = False

    return result


# Helper functions

def _generate_random_points(
    rng: random.Random,
    params: Dict[str, Any],
    point_count: int = 10,
) -> List[tuple]:
    """
    Generate random stroke points.

    Args:
        rng: Random instance (seeded for determinism)
        params: Generation parameters
        point_count: Number of points to generate

    Returns:
        List of (x, y, z, pressure, strength) tuples
    """
    points = []
    for i in range(point_count):
        x = rng.uniform(-1, 1)
        y = rng.uniform(-1, 1)
        z = rng.uniform(0, 0.1)  # Slight Z variation
        pressure = rng.uniform(0.8, 1.0)
        strength = 1.0
        points.append((x, y, z, pressure, strength))
    return points
