"""
GP Material system for Grease Pencil.

Provides material creation and configuration for GP strokes
and fills using MaterialGPencilStyle and shader node groups.
"""

from __future__ import annotations

from typing import TYPE_CHECKING, Any, Dict, List, Optional, Tuple

if TYPE_CHECKING:
    from bpy.types import Material

from .types import (
    GPMaterialConfig,
    StrokeStyle,
    FillStyle,
    StrokeMode,
)


def create_gp_material(
    name: str,
    config: Optional[GPMaterialConfig] = None,
    **kwargs: Any,
) -> Material:
    """
    Create a Grease Pencil material.

    Args:
        name: Material name
        config: Optional GPMaterialConfig for configuration
        **kwargs: Override any config values

    Returns:
        The created material with GP settings

    Raises:
        ImportError: If bpy not available

    Example:
        >>> mat = create_gp_material(
        ...     "outline_material",
        ...     stroke_style=StrokeStyle.SOLID,
        ...     show_stroke=True,
        ...     show_fill=False,
        ... )
    """
    try:
        import bpy
    except ImportError as exc:
        raise ImportError(
            "create_gp_material requires Blender (bpy module)."
        ) from exc

    # Create material
    mat = bpy.data.materials.new(name)
    mat.use_nodes = True

    # Get GP material settings
    gp_mat = mat.grease_pencil

    # Apply config if provided
    if config:
        gp_mat.show_stroke = config.show_stroke
        gp_mat.show_fill = config.show_fill
        gp_mat.color = config.color
        gp_mat.fill_color = config.fill_color
        gp_mat.mode = config.stroke_mode.value
        gp_mat.stroke_style = config.stroke_style.value
        gp_mat.fill_style = config.fill_style.value
        gp_mat.use_stroke_holdout = config.use_stroke_holdout
        gp_mat.use_fill_holdout = config.use_fill_holdout

        # Set overlap mode
        if hasattr(gp_mat, 'overlap_mode'):
            gp_mat.overlap_mode = config.overlap_mode

        # Set mix factor for gradients
        if config.fill_style == FillStyle.GRADIENT:
            if hasattr(gp_mat, 'mix_factor'):
                gp_mat.mix_factor = config.mix_factor

    # Apply any kwargs overrides
    for key, value in kwargs.items():
        if hasattr(gp_mat, key):
            setattr(gp_mat, key, value)

    return mat


def apply_material_to_gp(
    gp_obj: Any,
    material: Material,
    slot_index: Optional[int] = None,
) -> int:
    """
    Apply a material to a GP object.

    Args:
        gp_obj: The GP object
        material: Material to apply
        slot_index: Optional slot index (appends if None)

    Returns:
        The material slot index

    Example:
        >>> idx = apply_material_to_gp(gp_obj, outline_mat)
    """
    if slot_index is not None:
        # Use existing slot
        if slot_index < len(gp_obj.material_slots):
            gp_obj.material_slots[slot_index].material = material
        return slot_index
    else:
        # Append new slot
        gp_obj.data.materials.append(material)
        return len(gp_obj.data.materials) - 1


def create_material_node_group(
    name: str,
    node_type: str,
    inputs: Dict[str, Any],
    outputs: List[str],
) -> Any:
    """
    Create a shader node group for GP materials.

    This follows the NodeTreeBuilder pattern from lib/geometry_nodes.

    Args:
        name: Node group name
        node_type: Type of node group (e.g., 'NPR_AnimeCel', 'GP_Stroke_Gradient')
        inputs: Input socket definitions with defaults
        outputs: Output socket names

    Returns:
        The created node group

    Note:
        This creates shader node groups, not geometry node groups.
        GP materials use shader nodes for NPR effects.
    """
    try:
        import bpy
    except ImportError as exc:
        raise ImportError(
            "create_material_node_group requires Blender (bpy module)."
        ) from exc

    # Remove existing group with same name
    if name in bpy.data.node_groups:
        bpy.data.node_groups.remove(bpy.data.node_groups[name])

    # Create shader node group
    node_group = bpy.data.node_groups.new(name, 'ShaderNodeTree')

    # Create input interface
    for input_name, input_def in inputs.items():
        if isinstance(input_def, dict):
            socket = node_group.inputs.new(
                input_def.get('type', 'RGBA'),
                input_name
            )
            if 'default' in input_def:
                socket.default_value = input_def['default']
        else:
            node_group.inputs.new('RGBA', input_name)

    # Create output interface
    for output_name in outputs:
        node_group.outputs.new('RGBA', output_name)

    return node_group


def create_outline_material(
    color: Tuple[float, float, float, float] = (0.0, 0.0, 0.0, 1.0),
    stroke_width: float = 3.0,
) -> Material:
    """
    Create a simple outline material.

    Convenience function for common outline use case.

    Args:
        color: Stroke color (RGBA)
        stroke_width: Stroke width

    Returns:
        The created outline material
    """
    return create_gp_material(
        "GP_Outline",
        stroke_style=StrokeStyle.SOLID,
        fill_style=FillStyle.SOLID,
        show_stroke=True,
        show_fill=False,
        color=color,
    )


def create_fill_material(
    fill_color: Tuple[float, float, float, float] = (1.0, 1.0, 1.0, 1.0),
) -> Material:
    """
    Create a simple fill material.

    Convenience function for common fill use case.

    Args:
        fill_color: Fill color (RGBA)

    Returns:
        The created fill material
    """
    return create_gp_material(
        "GP_Fill",
        stroke_style=StrokeStyle.SOLID,
        fill_style=FillStyle.SOLID,
        show_stroke=False,
        show_fill=True,
        fill_color=fill_color,
    )


def create_gradient_material(
    color_start: Tuple[float, float, float, float],
    color_end: Tuple[float, float, float, float],
    gradient_type: str = "LINEAR",
) -> Material:
    """
    Create a gradient fill material.

    Args:
        color_start: Start color (RGBA)
        color_end: End color (RGBA)
        gradient_type: Gradient type (LINEAR, RADIAL, etc.)

    Returns:
        The created gradient material
    """
    try:
        import bpy
    except ImportError as exc:
        raise ImportError(
            "create_gradient_material requires Blender (bpy module)."
        ) from exc

    mat = create_gp_material(
        "GP_Gradient",
        stroke_style=StrokeStyle.SOLID,
        fill_style=FillStyle.GRADIENT,
        show_stroke=True,
        show_fill=True,
        fill_color=color_start,  # Start color
    )

    # Configure gradient (API varies by Blender version)
    gp_mat = mat.grease_pencil
    if hasattr(gp_mat, 'mix_color'):
        gp_mat.mix_color = color_end
    if hasattr(gp_mat, 'gradient_type'):
        gp_mat.gradient_type = gradient_type

    return mat
