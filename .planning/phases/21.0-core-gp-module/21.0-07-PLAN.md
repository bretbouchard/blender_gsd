```python
"""
Brush configuration system for Grease Pencil.

Provides brush presets for artist-assisted drawing workflows.
These are NOT procedural generation tools - they configure
the brush tool for manual drawing.

Usage:
    preset = load_brush_preset("pencil_soft")
    apply_brush_preset(preset)
"""

from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Dict, List, Optional, Tuple

import yaml


@dataclass
class BrushPreset:
    """
    Configuration for a GP brush preset.

    These settings configure the brush tool for manual drawing,
    not procedural stroke generation.
    """

    id: str
    name: str
    description: str = ""

    # Stroke settings
    stroke_width: float = 10.0
    strength: float = 1.0
    hardness: float = 1.0

    # Pressure sensitivity
    use_pressure_strength: bool = False
    use_pressure_thickness: bool = True
    use_pressure_opacity: bool = False

    # Stroke curve settings
    curve_x: List[float] = field(default_factory=lambda: [0.0, 1.0])
    curve_y: List[float] = field(default_factory=lambda: [0.0, 1.0])

    # Input settings
    use_stabilizer: bool = True
    stabilizer_factor: float = 0.5
    stabilizer_radius: int = 32

    # Color settings (for color-enabled brushes)
    color: Tuple[float, float, float, float] = (0.0, 0.0, 0.0, 1.0)
    secondary_color: Tuple[float, float, float, float] = (1.0, 1.0, 1.0, 1.0)

    # Brush type
    brush_type: str = "PENCIL"  # PENCIL, PEN, MARKER, AIRBRUSH, FILL, ERASE

    # Material settings
    material_index: int = 0

    # Display settings
    show_line_jitter: bool = False
    jitter_amount: float = 0.0

    def to_dict(self) -> Dict[str, Any]:
        """Convert to dictionary for serialization."""
        return {
            'id': self.id,
            'name': self.name,
            'description': self.description,
            'stroke_width': self.stroke_width,
            'strength': self.strength,
            'hardness': self.hardness,
            'use_pressure_strength': self.use_pressure_strength,
            'use_pressure_thickness': self.use_pressure_thickness,
            'use_pressure_opacity': self.use_pressure_opacity,
            'curve_x': self.curve_x,
            'curve_y': self.curve_y,
            'use_stabilizer': self.use_stabilizer,
            'stabilizer_factor': self.stabilizer_factor,
            'stabilizer_radius': self.stabilizer_radius,
            'color': self.color,
            'secondary_color': self.secondary_color,
            'brush_type': self.brush_type,
            'material_index': self.material_index,
            'show_line_jitter': self.show_line_jitter,
            'jitter_amount': self.jitter_amount,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> BrushPreset:
        """Create from dictionary."""
        return cls(
            id=data.get('id', 'unknown'),
            name=data.get('name', 'Unknown Brush'),
            description=data.get('description', ''),
            stroke_width=data.get('stroke_width', 10.0),
            strength=data.get('strength', 1.0),
            hardness=data.get('hardness', 1.0),
            use_pressure_strength=data.get('use_pressure_strength', False),
            use_pressure_thickness=data.get('use_pressure_thickness', True),
            use_pressure_opacity=data.get('use_pressure_opacity', False),
            curve_x=data.get('curve_x', [0.0, 1.0]),
            curve_y=data.get('curve_y', [0.0, 1.0]),
            use_stabilizer=data.get('use_stabilizer', True),
            stabilizer_factor=data.get('stabilizer_factor', 0.5),
            stabilizer_radius=data.get('stabilizer_radius', 32),
            color=tuple(data.get('color', (0.0, 0.0, 0.0, 1.0))),
            secondary_color=tuple(data.get('secondary_color', (1.0, 1.0, 1.0, 1.0))),
            brush_type=data.get('brush_type', 'PENCIL'),
            material_index=data.get('material_index', 0),
            show_line_jitter=data.get('show_line_jitter', False),
            jitter_amount=data.get('jitter_amount', 0.0),
        )


# Preset cache
_preset_cache: Optional[Dict[str, BrushPreset]] = None


def _load_presets() -> Dict[str, BrushPreset]:
    """Load brush presets from YAML file."""
    global _preset_cache

    if _preset_cache is not None:
        return _preset_cache

    preset_path = Path(__file__).parent.parent.parent / "configs" / "grease_pencil" / "brush_presets.yaml"

    if not preset_path.exists():
        _preset_cache = {}
        return _preset_cache

    with open(preset_path, 'r') as f:
        data = yaml.safe_load(f) or {}

    _preset_cache = {
        preset_id: BrushPreset.from_dict({**preset_data, 'id': preset_id})
        for preset_id, preset_data in data.items()
        if isinstance(preset_data, dict)
    }

    return _preset_cache


def load_brush_preset(preset_id: str) -> Optional[BrushPreset]:
    """
    Load a brush preset by ID.

    Args:
        preset_id: The preset identifier

    Returns:
        BrushPreset if found, None otherwise

    Example:
        >>> preset = load_brush_preset("pencil_soft")
        >>> if preset:
        ...     print(preset.name)
    """
    presets = _load_presets()
    return presets.get(preset_id)


def list_brush_presets() -> List[str]:
    """
    List all available brush preset IDs.

    Returns:
        List of preset identifiers

    Example:
        >>> presets = list_brush_presets()
        >>> print(presets)
        ['pencil_soft', 'pencil_hard', 'pen_outline', ...]
    """
    presets = _load_presets()
    return sorted(presets.keys())


def get_brush_presets_by_type(brush_type: str) -> List[BrushPreset]:
    """
    Get all presets of a specific brush type.

    Args:
        brush_type: Type to filter by (PENCIL, PEN, MARKER, etc.)

    Returns:
        List of matching presets
    """
    presets = _load_presets()
    return [p for p in presets.values() if p.brush_type == brush_type]


def apply_brush_preset(preset: BrushPreset) -> None:
    """
    Apply a brush preset to the current brush tool.

    This configures the active GP brush tool for manual drawing.
    Requires Blender context.

    Args:
        preset: The brush preset to apply

    Raises:
        ImportError: If bpy not available
    """
    try:
        import bpy
    except ImportError as exc:
        raise ImportError(
            "apply_brush_preset requires Blender (bpy module)."
        ) from exc

    # Get the active GP brush
    brush = bpy.context.tool_settings.gpencil_paint.brush

    if brush is None:
        # Create new brush if none exists
        brush = bpy.data.brushes.new(preset.name, mode='GPENCIL_PAINT')
        bpy.context.tool_settings.gpencil_paint.brush = brush

    # Apply settings
    brush.size = preset.stroke_width
    brush.strength = preset.strength
    brush.gpencil_settings.pen_hardness = preset.hardness

    # Pressure settings
    brush.gpencil_settings.use_pressure_strength = preset.use_pressure_strength
    brush.gpencil_settings.use_pressure_thickness = preset.use_pressure_thickness

    # Stabilizer settings
    brush.use_smooth_stroke = preset.use_stabilizer
    brush.smooth_stroke_factor = preset.stabilizer_factor
    brush.smooth_stroke_radius = preset.stabilizer_radius

    # Color
    brush.gpencil_settings.color = preset.color

    # Curve
    if preset.curve_x and preset.curve_y:
        brush.curve_preset = 'CUSTOM'
        # Reset and create curve points
        brush.curve.curves[0].points.clear()
        for i, (x, y) in enumerate(zip(preset.curve_x, preset.curve_y)):
            if i == 0:
                brush.curve.curves[0].points[0].location = (x, y, 0)
            else:
                brush.curve.curves[0].points.new((x, y, 0))


def create_custom_preset(
    name: str,
    stroke_width: float = 10.0,
    hardness: float = 1.0,
    brush_type: str = "PENCIL",
    **kwargs: Any,
) -> BrushPreset:
    """
    Create a custom brush preset.

    Args:
        name: Preset name
        stroke_width: Stroke width
        hardness: Stroke hardness
        brush_type: Brush type (PENCIL, PEN, MARKER, etc.)
        **kwargs: Additional brush settings

    Returns:
        New BrushPreset instance
    """
    return BrushPreset(
        id=name.lower().replace(" ", "_"),
        name=name,
        stroke_width=stroke_width,
        hardness=hardness,
        brush_type=brush_type,
        **kwargs,
    )
```

Key distinction: BrushPreset is for MANUAL drawing tools, not procedural generation. The generate_strokes_procedural function in stroke_utils.py is for programmatic stroke creation.
