    """
    Create a Grease Pencil object with layers.

    Uses Blender 4.3+ GP v3 API.

    Args:
        name: Name for the GP object
        layers: Optional layer configurations

    Returns:
        The created GP object

    Raises:
        ImportError: If bpy not available (not in Blender)
    """

        import bpy

def generate_strokes_procedural(
    params: Dict[str, Any],
    seed: Optional[int] = None,
) -> List[GPStrokeConfig]:
    """
    Generate strokes procedurally with deterministic output.

    Args:
        params: Generation parameters including:
            - stroke_count: Number of strokes to generate
            - stroke_width: Base stroke width
            - points_per_stroke: Points per stroke
            - layer_names: Names for layers
        seed: Optional seed for determinism

    Returns:
        List of GPStrokeConfig objects

    Example:
        >>> strokes = generate_strokes_procedural(
        ...     {'stroke_count': 5, 'stroke_width': 3.0},
        ...     seed=42
        ... )
        >>> len(strokes)
        5
    """
    import random

    from .utils import generate_seed_from_params

    # Derive seed if not provided
    if seed is None:
        seed = generate_seed_from_params(params)

    # Log for reproducibility debugging
    print(f"[GP] Using seed: {seed}")

    # Set random state
    rng = random.Random(seed)

    # Get parameters with defaults
    stroke_count = params.get('stroke_count', 10)
    stroke_width = params.get('stroke_width', 5.0)
    points_per_stroke = params.get('points_per_stroke', 10)

    # Generate strokes
    strokes: List[GPStrokeConfig] = []
    for i in range(stroke_count):
        points: List[Tuple[float, float, float, float, float]] = []
        for j in range(points_per_stroke):
            x = rng.uniform(-1, 1)
            y = rng.uniform(-1, 1)
            z = rng.uniform(0, 0.1)
            pressure = rng.uniform(0.8, 1.0)
            strength = 1.0
            points.append((x, y, z, pressure, strength))

        stroke = GPStrokeConfig(
            id=f"stroke_{i:04d}",
            points=points,
            stroke_width=stroke_width,
            hardness=1.0,
            use_cyclic=False,
            material_index=0,
            display_mode=DisplayMode.D3SPACE,
            seed=rng.randint(0, 2**31),
        )
        strokes.append(stroke)

    return strokes


def manipulate_stroke_points(
    config: GPStrokeConfig,
    transformation: str,
    params: Dict[str, Any],
) -> GPStrokeConfig:
    """
    Apply transformation to stroke points.

    Args:
        config: Stroke configuration to transform
        transformation: Type of transformation:
            - 'scale': Scale points by factor
            - 'rotate': Rotate points by angle
            - 'translate': Offset points by vector
            - 'noise': Add noise to points
        params: Transformation parameters

    Returns:
        New GPStrokeConfig with transformed points

    Example:
        >>> scaled = manipulate_stroke_points(
        ...     config,
        ...     'scale',
        ...     {'factor': 2.0}
        ... )
    """
    import math
    import random

    points = list(config.points)  # Copy points

    if transformation == 'scale':
        factor = params.get('factor', 1.0)
        points = [
            (p[0] * factor, p[1] * factor, p[2] * factor, p[3], p[4])
            for p in points
        ]
    elif transformation == 'rotate':
        angle = params.get('angle', 0.0)
        cos_a = math.cos(angle)
        sin_a = math.sin(angle)
        points = [
            (p[0] * cos_a - p[1] * sin_a, p[0] * sin_a + p[1] * cos_a, p[2], p[3], p[4])
            for p in points
        ]
    elif transformation == 'translate':
        offset = params.get('offset', (0, 0, 0))
        points = [
            (p[0] + offset[0], p[1] + offset[1], p[2] + offset[2], p[3], p[4])
            for p in points
        ]
    elif transformation == 'noise':
        amount = params.get('amount', 0.1)
        rng = random.Random(config.seed)
        points = [
            (p[0] + rng.uniform(-amount, amount),
             p[1] + rng.uniform(-amount, amount),
             p[2] + rng.uniform(-amount * 0.1, amount * 0.1),
             p[3], p[4])
            for p in points
        ]

    return GPStrokeConfig(
        id=f"{config.id}_transformed",
        points=points,
        stroke_width=config.stroke_width,
        hardness=config.hardness,
        use_cyclic=config.use_cyclic,
        material_index=config.material_index,
        display_mode=config.display_mode,
        seed=config.seed,
    )
