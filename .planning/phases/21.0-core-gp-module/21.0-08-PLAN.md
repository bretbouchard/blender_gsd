"""
Unified preset loader for Grease Pencil system.

Provides a single interface to load all preset types:
- Stroke presets
- Material presets
- Brush presets
- Modifier presets
- Effect presets
- Animation presets
"""

from __future__ import annotations

from enum import Enum
from pathlib import Path
from typing import Any, Dict, List, Optional, Union

import yaml


class PresetType(Enum):
    """Types of GP presets."""
    STROKE = "stroke"
    MATERIAL = "material"
    BRUSH = "brush"
    MODIFIER = "modifier"
    EFFECT = "effect"
    ANIMATION = "animation"


# Preset directory mapping
_PRESET_FILES = {
    PresetType.STROKE: "stroke_presets.yaml",
    PresetType.MATERIAL: "material_presets.yaml",
    PresetType.BRUSH: "brush_presets.yaml",
    PresetType.MODIFIER: "modifier_presets.yaml",
    PresetType.EFFECT: "effect_presets.yaml",
    PresetType.ANIMATION: "animation_presets.yaml",
}

# Preset cache
_preset_cache: Dict[PresetType, Dict[str, Any]] = {}


def _get_preset_path(preset_type: PresetType) -> Path:
    """Get path to preset file."""
    return Path(__file__).parent.parent.parent / "configs" / "grease_pencil" / _PRESET_FILES[preset_type]


def _load_preset_file(preset_type: PresetType) -> Dict[str, Any]:
    """Load presets from YAML file."""
    if preset_type in _preset_cache:
        return _preset_cache[preset_type]

    preset_path = _get_preset_path(preset_type)

    if not preset_path.exists():
        _preset_cache[preset_type] = {}
        return _preset_cache[preset_type]

    with open(preset_path, 'r') as f:
        data = yaml.safe_load(f) or {}

    _preset_cache[preset_type] = data
    return data


def load_preset(
    preset_type: Union[PresetType, str],
    preset_id: str,
) -> Optional[Dict[str, Any]]:
    """
    Load a preset by type and ID.

    Args:
        preset_type: Type of preset (PresetType enum or string)
        preset_id: Preset identifier

    Returns:
        Preset dictionary if found, None otherwise

    Example:
        >>> preset = load_preset("stroke", "basic_line")
        >>> print(preset['stroke_width'])
        2.0
    """
    if isinstance(preset_type, str):
        preset_type = PresetType(preset_type.lower())

    presets = _load_preset_file(preset_type)
    return presets.get(preset_id)


def list_presets(preset_type: Union[PresetType, str]) -> List[str]:
    """
    List all preset IDs for a type.

    Args:
        preset_type: Type of preset

    Returns:
        List of preset identifiers

    Example:
        >>> list_presets("stroke")
        ['basic_line', 'thick_outline', 'soft_sketch', ...]
    """
    if isinstance(preset_type, str):
        preset_type = PresetType(preset_type.lower())

    presets = _load_preset_file(preset_type)
    return sorted(presets.keys())


def get_all_presets() -> Dict[str, List[str]]:
    """
    Get all presets organized by type.

    Returns:
        Dictionary mapping preset type names to lists of preset IDs
    """
    result = {}
    for preset_type in PresetType:
        result[preset_type.value] = list_presets(preset_type)
    return result


def clear_cache() -> None:
    """Clear the preset cache."""
    global _preset_cache
    _preset_cache = {}


def reload_presets() -> None:
    """Reload all presets from disk."""
    clear_cache()
    for preset_type in PresetType:
        _load_preset_file(preset_type)


def validate_preset(
    preset_type: Union[PresetType, str],
    preset_id: str,
    required_fields: List[str],
) -> bool:
    """
    Validate that a preset has required fields.

    Args:
        preset_type: Type of preset
        preset_id: Preset identifier
        required_fields: List of field names that must exist

    Returns:
        True if valid, False otherwise
    """
    preset = load_preset(preset_type, preset_id)
    if preset is None:
        return False

    return all(field in preset for field in required_fields)

        assert restored.id == original.id
        assert restored.stroke_width == original.stroke_width

        """Test GPMaterialConfig creation."""

            id="test_material",

        seed1 = generate_seed_from_params(params)
        seed2 = generate_seed_from_params(params)
        assert seed1 == seed2

    def test_seed_generation_different_params(self):
        """Test that different params produce different seeds."""
        seed1 = generate_seed_from_params({'count': 10})
        seed2 = generate_seed_from_params({'count': 20})
        assert seed1 != seed2

    def test_procedural_generation_deterministic(self):
        """Test that procedural generation is deterministic."""
        params = {'stroke_count': 5, 'stroke_width': 3.0}
        strokes1 = generate_strokes_procedural(params, seed=42)
        strokes2 = generate_strokes_procedural(params, seed=42)
        assert len(strokes1) == len(strokes2)
        assert len(strokes1) == 5
        # Same seed should produce identical results
        for s1, s2 in zip(strokes1, strokes2):
            assert s1.points == s2.points
