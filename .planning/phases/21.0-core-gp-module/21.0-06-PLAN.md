```python
"""
GP Effects - High-level effect presets for Grease Pencil.

Provides user-friendly effect functions that wrap GP modifiers
with sensible defaults and preset configurations.

IMPORTANT: Effects use GP modifiers (GP_BUILD, GP_NOISE, etc.)
NOT Geometry Nodes. Geometry Nodes cannot process GP data.
"""

from __future__ import annotations

from typing import TYPE_CHECKING, Any, Dict, Optional

if TYPE_CHECKING:
    from bpy.types import GreasePencilModifier, Object as BlenderObject

from .types import GPMaskConfig
from .modifiers import (
    apply_build_modifier,
    apply_noise_modifier,
    apply_smooth_modifier,
    apply_opacity_modifier,
    apply_color_modifier,
)


def apply_build_effect(
    gp_obj: BlenderObject,
    effect_type: str = "reveal",
    frame_start: int = 1,
    frame_duration: int = 50,
    seed: Optional[int] = None,
    mask_config: Optional[GPMaskConfig] = None,
) -> GreasePencilModifier:
    """
    Apply build effect for draw-on animation.

    Higher-level wrapper around GP_BUILD modifier with preset options.

    Args:
        gp_obj: GP object to modify
        effect_type: Effect preset type:
            - "reveal": Sequential stroke reveal
            - "concurrent": All strokes appear together
            - "additive": Strokes appear one by one
            - "fade": Fade in effect
        frame_start: Start frame
        frame_duration: Duration in frames
        seed: Random seed for determinism
        mask_config: Optional mask configuration

    Returns:
        The created GP_BUILD modifier

    Example:
        >>> mod = apply_build_effect(
        ...     gp_obj,
        ...     effect_type="reveal",
        ...     frame_duration=24,
        ... )
    """
    # Map effect types to modifier settings
    effect_configs = {
        "reveal": {
            "mode": "SEQUENTIAL",
            "transition": "GROW",
        },
        "concurrent": {
            "mode": "CONCURRENT",
            "transition": "GROW",
        },
        "additive": {
            "mode": "ADDITIVE",
            "transition": "GROW",
        },
        "fade": {
            "mode": "SEQUENTIAL",
            "transition": "FADE",
        },
    }

    config = effect_configs.get(effect_type, effect_configs["reveal"])

    return apply_build_modifier(
        gp_obj,
        name=f"Build_{effect_type.capitalize()}",
        mode=config["mode"],
        transition=config["transition"],
        frame_start=frame_start,
        frame_duration=frame_duration,
        seed=seed,
        mask_config=mask_config,
    )


def apply_noise_effect(
    gp_obj: BlenderObject,
    effect_type: str = "wiggle",
    intensity: float = 0.5,
    scale: float = 1.0,
    seed: Optional[int] = None,
    mask_config: Optional[GPMaskConfig] = None,
) -> GreasePencilModifier:
    """
    Apply noise effect for procedural wiggle/distortion.

    Higher-level wrapper around GP_NOISE modifier with preset options.

    Args:
        gp_obj: GP object to modify
        effect_type: Effect preset type:
            - "wiggle": Light organic movement
            - "shake": Heavy shake effect
            - "jitter": Random jitter
        intensity: Effect intensity (0-1)
        scale: Noise scale
        seed: Random seed for determinism
        mask_config: Optional mask configuration

    Returns:
        The created GP_NOISE modifier

    Example:
        >>> mod = apply_noise_effect(
        ...     gp_obj,
        ...     effect_type="wiggle",
        ...     intensity=0.3,
        ... )
    """
    effect_configs = {
        "wiggle": {"factor": 0.3, "scale": 1.0, "noise_mode": "ADD"},
        "shake": {"factor": 0.7, "scale": 2.0, "noise_mode": "ADD"},
        "jitter": {"factor": 0.5, "scale": 0.5, "noise_mode": "ADD"},
    }

    config = effect_configs.get(effect_type, effect_configs["wiggle"])

    # Override with provided intensity and scale
    config["factor"] = intensity
    config["scale"] = scale

    return apply_noise_modifier(
        gp_obj,
        name=f"Noise_{effect_type.capitalize()}",
        factor=config["factor"],
        scale=config["scale"],
        seed=seed,
        noise_mode=config["noise_mode"],
        mask_config=mask_config,
    )


def apply_smooth_effect(
    gp_obj: BlenderObject,
    effect_type: str = "light",
    mask_config: Optional[GPMaskConfig] = None,
) -> GreasePencilModifier:
    """
    Apply smooth effect for stroke smoothing.

    Higher-level wrapper around GP_SMOOTH modifier with preset options.

    Args:
        gp_obj: GP object to modify
        effect_type: Effect preset type:
            - "light": Light smoothing
            - "medium": Medium smoothing
            - "heavy": Heavy smoothing
        mask_config: Optional mask configuration

    Returns:
        The created GP_SMOOTH modifier

    Example:
        >>> mod = apply_smooth_effect(gp_obj, effect_type="medium")
    """
    effect_configs = {
        "light": {"factor": 0.3, "steps": 1},
        "medium": {"factor": 0.5, "steps": 2},
        "heavy": {"factor": 0.8, "steps": 3},
    }

    config = effect_configs.get(effect_type, effect_configs["light"])

    return apply_smooth_modifier(
        gp_obj,
        name=f"Smooth_{effect_type.capitalize()}",
        factor=config["factor"],
        steps=config["steps"],
        mask_config=mask_config,
    )


def apply_fade_effect(
    gp_obj: BlenderObject,
    opacity: float = 1.0,
    mask_config: Optional[GPMaskConfig] = None,
) -> GreasePencilModifier:
    """
    Apply fade effect for opacity control.

    Higher-level wrapper around GP_OPACITY modifier.

    Args:
        gp_obj: GP object to modify
        opacity: Opacity factor (0-1)
            - 1.0: Full opacity
            - 0.5: Half transparent
            - 0.3: Ghost/trace effect
        mask_config: Optional mask configuration

    Returns:
        The created GP_OPACITY modifier

    Example:
        >>> mod = apply_fade_effect(gp_obj, opacity=0.5)
    """
    effect_name = "Fade_Full" if opacity >= 1.0 else f"Fade_{int(opacity*100)}"

    return apply_opacity_modifier(
        gp_obj,
        name=effect_name,
        factor=opacity,
        mask_config=mask_config,
    )


def apply_color_shift_effect(
    gp_obj: BlenderObject,
    effect_type: str = "warm",
    mask_config: Optional[GPMaskConfig] = None,
) -> GreasePencilModifier:
    """
    Apply color shift effect for color grading.

    Higher-level wrapper around GP_COLOR modifier with preset options.

    Args:
        gp_obj: GP object to modify
        effect_type: Effect preset type:
            - "warm": Warm color shift
            - "cool": Cool color shift
            - "desaturate": Reduce saturation
            - "boost": Increase saturation
        mask_config: Optional mask configuration

    Returns:
        The created GP_COLOR modifier

    Example:
        >>> mod = apply_color_shift_effect(gp_obj, effect_type="warm")
    """
    effect_configs = {
        "warm": {"hue": 0.05, "saturation": 1.2, "value": 1.0},
        "cool": {"hue": -0.05, "saturation": 1.0, "value": 1.0},
        "desaturate": {"hue": 0.0, "saturation": 0.5, "value": 1.0},
        "boost": {"hue": 0.0, "saturation": 1.5, "value": 1.0},
    }

    config = effect_configs.get(effect_type, effect_configs["warm"])

    return apply_color_modifier(
        gp_obj,
        name=f"Color_{effect_type.capitalize()}",
        hue=config["hue"],
        saturation=config["saturation"],
        value=config["value"],
        mask_config=mask_config,
    )


def apply_anime_effect_stack(
    gp_obj: BlenderObject,
    preset: str = "default",
    seed: Optional[int] = None,
) -> list:
    """
    Apply a complete anime-style effect stack.

    Combines multiple effects for common anime/NPR looks.

    Args:
        gp_obj: GP object to modify
        preset: Preset name:
            - "default": Standard anime look
            - "retro": 90s anime aesthetic
            - "watercolor": Soft watercolor look
        seed: Random seed for determinism

    Returns:
        List of created modifiers

    Example:
        >>> mods = apply_anime_effect_stack(gp_obj, preset="default")
    """
    modifiers = []

    if preset == "default":
        # Standard anime: light smooth + outline
        modifiers.append(apply_smooth_effect(gp_obj, "light"))
        modifiers.append(apply_build_effect(gp_obj, "reveal", seed=seed))

    elif preset == "retro":
        # 90s anime: more noise, less smooth
        modifiers.append(apply_noise_effect(gp_obj, "wiggle", intensity=0.2, seed=seed))
        modifiers.append(apply_color_shift_effect(gp_obj, "desaturate"))

    elif preset == "watercolor":
        # Watercolor: heavy smooth, fade
        modifiers.append(apply_smooth_effect(gp_obj, "heavy"))
        modifiers.append(apply_fade_effect(gp_obj, opacity=0.9))

    return modifiers


def clear_all_effects(gp_obj: BlenderObject) -> None:
    """
    Remove all GP modifiers from object.

    Args:
        gp_obj: GP object to clear effects from
    """
    # Remove all GP modifiers
    for modifier in list(gp_obj.grease_pencil_modifiers):
        gp_obj.grease_pencil_modifiers.remove(modifier)
```

This module provides a higher-level, more user-friendly API than the raw modifiers.py. Think of it as "presets on top of modifiers".
